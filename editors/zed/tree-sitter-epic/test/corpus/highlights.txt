    left: i64

type Expr = Lit | Add

fun (p: Parser) peek(offset: i64): Expr {
    ret new Expr(new Lit { value: p.name })
}

fun demo(flag: bool, parser: Parser): void {
    let xs: i64[] = new i64[] { 1, 2, 3 }
    let buf = new u8[16]
    let text = f"name={parser.name} len={len(xs)}"
    let view = text[0:len(text)]

    for flag {
        xs[0] += 1
        break
    }

    for i: 0:len(xs) {
        if i == 1 {
            continue
        } else if parser.name? {
            parser.pos = parser.pos + xs[i]
        }
    }

    match xs[0] {
        0: { println("zero") }
        _: { println(text) }
    }

    let expr: Expr = new Expr(new Add { left: 20, right: 22 })
    match expr {
        Lit lit: { println(lit.value) }
        Add add: { println(str(add.left + add.right)) }
        _: { panic "unreachable" }
    }

    buf.push('A')
    if parser.name? {
        println(view)
    }
    ret
}
